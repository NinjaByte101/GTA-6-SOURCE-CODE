3. Email Validation (Smart Login)

import 'package:flutter/material.dart';
void main() => runApp(const MaterialApp(debugShowCheckedModeBanner: false, home: LoginScreen()));
class LoginScreen extends StatefulWidget {
  const LoginScreen({super.key});
  @override
  State<LoginScreen> createState() => _LoginScreenState();
}
class _LoginScreenState extends State<LoginScreen> {
  final userCtrl = TextEditingController(), passCtrl = TextEditingController();

  
  bool isValidEmail(String email) => RegExp(r'^[\w.-]+@[\w.-]+\.\w+$').hasMatch(email);

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text('Login Screen'), centerTitle: true),
      body: Padding(
        padding: const EdgeInsets.all(20),
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            TextField(controller: userCtrl, decoration: const InputDecoration(labelText: 'Email', border: OutlineInputBorder())),
            const SizedBox(height: 20),
            TextField(controller: passCtrl, obscureText: true, decoration: const InputDecoration(labelText: 'Password', border: OutlineInputBorder())),
            const SizedBox(height: 30),
            ElevatedButton(
              onPressed: () {
                final isValid = isValidEmail(userCtrl.text) && passCtrl.text.isNotEmpty;
                ScaffoldMessenger.of(context).showSnackBar(SnackBar(
                  content: Text(isValid ? 'Welcome!' : !isValidEmail(userCtrl.text) ? 'Enter a valid email' : 'Please enter all fields'),
                  backgroundColor: isValid ? Colors.green : Colors.red,
                ));
              },
              child: const Text('Login'),
            ),
          ],
        ),
      ),
    );
  }
}

5. Counter with Auto Increment


import 'dart:async';
import 'package:flutter/material.dart';

void main() => runApp(const MaterialApp(debugShowCheckedModeBanner: false, home: CounterScreen()));

class CounterScreen extends StatefulWidget {
  const CounterScreen({super.key});
  @override
  State<CounterScreen> createState() => _CounterScreenState();
}

class _CounterScreenState extends State<CounterScreen> {
  int counter = 0;
  Timer? timer;

  void handleTimer({bool start = false, bool reset = false}) {
    timer?.cancel();
    timer = start ? Timer.periodic(const Duration(seconds: 1), (_) => setState(() => counter++)) : null;
    if (reset) setState(() => counter = 0);
  }

  @override
  void dispose() { timer?.cancel(); super.dispose(); }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text("Auto Increment Counter"), centerTitle: true),
      body: Column(
        mainAxisAlignment: MainAxisAlignment.center,
        children: [
          Text('$counter', style: const TextStyle(fontSize: 80, fontWeight: FontWeight.bold)),
          const SizedBox(height: 40),
          Row(
            mainAxisAlignment: MainAxisAlignment.spaceEvenly,
            children: [
              ElevatedButton(onPressed: () => handleTimer(start: true), child: const Text("Start")),
              ElevatedButton(onPressed: handleTimer, child: const Text("Pause")),
              ElevatedButton(onPressed: () => handleTimer(reset: true), child: const Text("Reset")),
            ],
          ),
        ],
      ),
    );
  }
}


6. Loading Screen (Progress Bar)


import 'dart:async';
import 'package:flutter/material.dart';

void main() => runApp(const MaterialApp(debugShowCheckedModeBanner: false, home: LoadingScreen()));

class LoadingScreen extends StatefulWidget {
  const LoadingScreen({super.key});
  @override
  State<LoadingScreen> createState() => _LoadingScreenState();
}

class _LoadingScreenState extends State<LoadingScreen> {
  double progress = 0.0;
  String message = "";
  Timer? timer;

  void startLoading() {
    setState(() { progress = 0.0; message = "Loading your awesome content..."; });
    timer?.cancel();
    timer = Timer.periodic(const Duration(milliseconds: 100), (t) {
      setState(() {
        progress += 0.01;
        if (progress >= 1.0) { progress = 1.0; message = "Finished! 🎉"; t.cancel(); }
      });
    });
  }

  @override
  void dispose() { timer?.cancel(); super.dispose(); }

  @override
  Widget build(BuildContext context) => Scaffold(
        appBar: AppBar(title: const Text("Loading Screen"), centerTitle: true),
        body: Padding(
          padding: const EdgeInsets.all(20),
          child: Column(
            mainAxisAlignment: MainAxisAlignment.center,
            children: [
              LinearProgressIndicator(value: progress, minHeight: 20, color: Colors.blue, backgroundColor: Colors.grey.shade300),
              const SizedBox(height: 20),
              Text(message, style: const TextStyle(fontSize: 18)),
              const SizedBox(height: 40),
              ElevatedButton(onPressed: startLoading, child: const Text("Start Loading")),
            ],
          ),
        ),
      );
}

8. Prime Number Finder

import 'package:flutter/material.dart';

void main() => runApp(MaterialApp(home: Scaffold(
  appBar: AppBar(title: Text('Prime Finder')),
  body: PrimeFinder(),
)));

class PrimeFinder extends StatefulWidget {
  @override
  State<PrimeFinder> createState() => _State();
}

class _State extends State<PrimeFinder> {
  final a = TextEditingController();
  final b = TextEditingController();
  String result = '';

  bool isPrime(int n) {
    if (n < 2) return false;
    for (int i = 2; i * i <= n; i++) if (n % i == 0) return false;
    return true;
  }

  Future<void> find() async {
    final lo = int.tryParse(a.text), hi = int.tryParse(b.text);
    if (lo == null || hi == null || lo > hi) return setState(() => result = 'Invalid input');
    List<int> primes = [];
    for (int i = lo; i <= hi; i++) {
      if (isPrime(i)) primes.add(i);
      if (i % 500 == 0) await Future.delayed(Duration.zero);
    }
    setState(() => result = primes.isEmpty ? 'No primes found' : primes.join(', '));
  }

  @override
  Widget build(BuildContext context) => Padding(
    padding: EdgeInsets.all(16),
    child: Column(children: [
      TextField(controller: a, keyboardType: TextInputType.number, decoration: InputDecoration(labelText: 'Lower')),
      TextField(controller: b, keyboardType: TextInputType.number, decoration: InputDecoration(labelText: 'Upper')),
      SizedBox(height: 12),
      ElevatedButton(onPressed: find, child: Text('Find Primes')),
      SizedBox(height: 12),
      Text(result),
    ]),
  );
}

14. Dark Mode Toggle (Shared Preferences)

import 'package:flutter/material.dart';
import 'package:shared_preferences/shared_preferences.dart';

void main() => runApp(const MyApp());

class MyApp extends StatefulWidget {
  const MyApp({super.key});
  @override
  State<MyApp> createState() => _MyAppState();
}

class _MyAppState extends State<MyApp> {
  bool isDark = false;

  @override
  void initState() { super.initState(); _load(); }

  Future<void> _load() async { 
    final p = await SharedPreferences.getInstance(); 
    setState(() => isDark = p.getBool('dark') ?? false); 
  }
  
  Future<void> _save(bool v) async => (await SharedPreferences.getInstance()).setBool('dark', v);

  @override
  Widget build(BuildContext context) => MaterialApp(
        debugShowCheckedModeBanner: false, 
        theme: isDark ? ThemeData.dark() : ThemeData.light(),
        home: Scaffold(
          appBar: AppBar(title: const Text('Dark Mode Toggle')),
          body: Center(child: Row(mainAxisAlignment: MainAxisAlignment.center, children: [
            const Text('Light'),
            Switch(value: isDark, onChanged: (v) { setState(() => isDark = v); _save(v); }),
            const Text('Dark'),
          ])),
        ),
      );
}
